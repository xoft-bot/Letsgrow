📊 Google Sheets → Firestore Sync
📋
How it works
Paste this Apps Script into your Google Sheet. Every time you save a change, it sends it to Firestore automatically. All club members see the update instantly.
// STEP 1: In your Google Sheet → Extensions → Apps Script // STEP 2: Replace YOUR_PROJECT_ID below // STEP 3: Deploy → New deployment → Web app → Anyone const FIREBASE_URL = 'https://firestore.googleapis.com/v1/projects/YOUR_PROJECT_ID/databases/(default)/documents/sheetsSync'; function onEdit(e) { const sheet = e.source.getActiveSheet().getName(); const row = e.range.getRow(); const col = e.range.getColumn(); const value = e.value; const member = e.source.getActiveSheet().getRange(row, 1).getValue(); const payload = { fields: { sheet: { stringValue: sheet }, row: { integerValue: row }, col: { integerValue: col }, value: { stringValue: String(value) }, member: { stringValue: String(member) }, timestamp: { stringValue: new Date().toISOString() } } }; UrlFetchApp.fetch(FIREBASE_URL, { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload) }); } // STEP 4: Set a time-based trigger: // Triggers → Add trigger → onEdit → On edit
🛡️ Firestore Security Rules
✅
Paste these rules in Firebase Console → Firestore → Rules
This ensures: only registered emails can read. Only admins can write. Members can only cast their own vote. No one can see another member's vote.
rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // Helper: is the user a registered member? function isRegistered() { return request.auth != null && exists(/databases/$(database)/documents/approvedEmails/$(request.auth.token.email)); } // Helper: is the user an admin? function isAdmin() { return isRegistered() && get(/databases/$(database)/documents/approvedEmails/$(request.auth.token.email)).data.role == 'admin'; } // Members and contributions: read by all registered, write by admin only match /members/{id} { allow read: if isRegistered(); allow write: if isAdmin(); } match /contributions/{id} { allow read: if isRegistered(); allow write: if isAdmin(); } match /committee/{id} { allow read: if isRegistered(); allow write: if isAdmin(); } // Approved emails: admin manages who can log in match /approvedEmails/{email} { allow read: if isRegistered(); allow write: if isAdmin(); } // Sessions: members manage only their own session match /sessions/{userId} { allow read, write: if request.auth.uid == userId || isAdmin(); } // Voting: open votes readable by all, ballots write-once per user match /votes/{voteId} { allow read: if isRegistered(); allow write: if isAdmin(); match /ballots/{userId} { // Can only vote for yourself, only once (create not update) allow create: if request.auth.uid == userId && isRegistered(); allow read: if isAdmin(); // only admin sees who voted what allow update, delete: if false; // votes cannot be changed or deleted } } // Sheets sync log: admin only match /sheetsSync/{id} { allow read, write: if isAdmin(); } } }
